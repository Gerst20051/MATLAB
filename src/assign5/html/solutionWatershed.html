clear all; close all;
load elevation.mat

% Choose Map
cmap = map;

% Display Map
imagesc(cmap); axis equal; colormap gray;

Finding pits

pits = findpits(cmap);
peaks = findpits(-cmap);
[num_pits,pitc] = size(pits);
[num_peaks,peakc] = size(peaks);
num_pits
num_peaks
imagesc(cmap); colormap(gray); axis equal
hold on
plot(pits(:,2),pits(:,1),'ro');
plot(peaks(:,2),peaks(:,1),'g+');
hold off
